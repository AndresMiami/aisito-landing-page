// Authentication Component with JWT Token Management
class AuthComponent {
    constructor() {
        this.authState = {
            isAuthenticated: false,
            user: null,
            token: null
        };
        this.eventBus = window.eventBus;
        this.setupEventListeners();
    }

    async login(credentials) {
        try {
            this.eventBus.emit('AUTH_LOGIN_START', { email: credentials.email });
            
            const response = await fetch('/api/auth/login', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify(credentials)
            });

            if (response.ok) {
                const data = await response.json();
                this.setAuthState(data.user, data.token);
                this.eventBus.emit('AUTH_VALIDATION_SUCCESS', { user: data.user });
                return { success: true, user: data.user };
            } else {
                throw new Error('Authentication failed');
            }
        } catch (error) {
            this.eventBus.emit('AUTH_ERROR', { error: error.message });
            return { success: false, error: error.message };
        }
    }

    setAuthState(user, token) {
        this.authState = { isAuthenticated: true, user, token };
        localStorage.setItem('authToken', token);
        this.eventBus.emit('SESSION_CREATED', { user });
    }
}

// Miami Experience Selector with Real-time Filtering
class ExperienceSelector {
    constructor(containerId) {
        this.container = document.getElementById(containerId);
        this.experiences = [];
        this.filters = { type: 'all', priceRange: 'all', location: 'all' };
        this.eventBus = window.eventBus;
        this.init();
    }

    async loadExperiences() {
        try {
            this.eventBus.emit('EXPERIENCE_SEARCH', { filters: this.filters });
            
            const response = await fetch('/api/experiences/miami');
            this.experiences = await response.json();
            
            this.renderExperiences();
            this.eventBus.emit('EXPERIENCES_LOADED', { count: this.experiences.length });
        } catch (error) {
            this.eventBus.emit('EXPERIENCE_ERROR', { error: error.message });
        }
    }

    applyFilters(newFilters) {
        this.filters = { ...this.filters, ...newFilters };
        this.eventBus.emit('FILTER_APPLIED', { filters: this.filters });
        
        const filtered = this.experiences.filter(exp => {
            return this.matchesFilters(exp, this.filters);
        });
        
        this.renderExperiences(filtered);
    }

    selectExperience(experienceId) {
        const experience = this.experiences.find(exp => exp.id === experienceId);
        this.eventBus.emit('EXPERIENCE_SELECTED', { experience });
        this.addToCart(experience);
    }
}

// Secure Payment Form with Stripe Integration
class PaymentForm {
    constructor() {
        this.stripe = Stripe(process.env.STRIPE_PUBLIC_KEY);
        this.elements = this.stripe.elements();
        this.eventBus = window.eventBus;
        this.setupForm();
    }

    setupForm() {
        this.cardElement = this.elements.create('card', {
            style: {
                base: { fontSize: '16px', color: '#424770' }
            }
        });
        this.cardElement.mount('#card-element');
    }

    async processPayment(bookingData) {
        try {
            this.eventBus.emit('PAYMENT_INIT', { amount: bookingData.total });
            
            const { token, error } = await this.stripe.createToken(this.cardElement);
            
            if (error) {
                throw new Error(error.message);
            }

            this.eventBus.emit('PAYMENT_VALIDATION', { token: token.id });
            
            const response = await fetch('/api/payments/process', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    token: token.id,
                    amount: bookingData.total,
                    booking: bookingData
                })
            });

            if (response.ok) {
                const result = await response.json();
                this.eventBus.emit('PAYMENT_SUCCESS', { transactionId: result.id });
                this.eventBus.emit('BOOKING_CONFIRMED', { booking: result.booking });
                return { success: true, booking: result.booking };
            }
        } catch (error) {
            this.eventBus.emit('PAYMENT_ERROR', { error: error.message });
            return { success: false, error: error.message };
        }
    }
}

// Centralized Event Bus for Component Communication
class EventBus {
    constructor() {
        this.events = new Map();
        this.debug = true;
    }

    on(eventName, callback) {
        if (!this.events.has(eventName)) {
            this.events.set(eventName, []);
        }
        this.events.get(eventName).push(callback);
        
        if (this.debug) {
            console.log(`EventBus: Registered listener for ${eventName}`);
        }
    }

    emit(eventName, data = {}) {
        if (this.debug) {
            console.log(`EventBus: Emitting ${eventName}`, data);
        }
        
        if (this.events.has(eventName)) {
            this.events.get(eventName).forEach(callback => {
                try {
                    callback(data);
                } catch (error) {
                    console.error(`EventBus: Error in ${eventName} listener:`, error);
                }
            });
        }
    }

    off(eventName, callback) {
        if (this.events.has(eventName)) {
            const callbacks = this.events.get(eventName);
            const index = callbacks.indexOf(callback);
            if (index > -1) {
                callbacks.splice(index, 1);
            }
        }
    }
}
